CREATE DATABASE APICURD
GO 
USE APICURD
GO 
CREATE TABLE EMPLOYEE
(
EMP_ID INT IDENTITY PRIMARY KEY,
EMP_NAME  VARCHAR(64)NULL,
EMP_AGE	  VARCHAR(64)NULL,
EMP_CITY  VARCHAR(64)NULL
)
GO

CREATE PROCEDURE  ACD_INSERT_UPDATE_EMPLOYEE
(
@EMP_ID INT=0,
@EMP_NAME   VARCHAR(64)=NULL,
@EMP_AGE    VARCHAR(64)=NULL,
@EMP_CITY   VARCHAR(64)=NULL
)
AS
BEGIN
    IF @EMP_ID > 0
	BEGIN
	  UPDATE EMPLOYEE
	  SET EMP_NAME=@EMP_NAME,
	      EMP_AGE=@EMP_AGE,
		  EMP_CITY=@EMP_CITY
	 WHERE EMP_ID=@EMP_ID
	END
	ELSE
	BEGIN
	   INSERT INTO EMPLOYEE
	         (
	         EMP_NAME,
			 EMP_AGE,
			 EMP_CITY
			 )
	    SELECT 
		    @EMP_NAME,
			@EMP_AGE,
			@EMP_CITY
	END
END
GO
CREATE PROCEDURE ACD_GET_ALL_EMPLOYEE
(
@EMP_ID INT=0,
@CAMMAND_TYPE VARCHAR(64)=NULL
)
AS
BEGIN
   IF @CAMMAND_TYPE = 'GET_ALL_EMPLOYEE'
   BEGIN
      SELECT * FROM EMPLOYEE
   END

   IF @CAMMAND_TYPE = 'GET_DATA_BY_ID'
   BEGIN
      SELECT * FROM EMPLOYEE WHERE EMP_ID=@EMP_ID
   END
END
GO
